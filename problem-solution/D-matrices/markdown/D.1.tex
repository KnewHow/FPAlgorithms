 b_{11}&   b_{12} & ... & b_{1n}\\

\\

 \sum_{d=1}^{n}a_{1d}b_{d1} & \sum_{d=1}^{n}a_{1d}b_{d2} & ... & \sum_{d=1}^{n}a_{1d}b_{dn}\\ \\
 \sum_{d=1}^{n}a_{2d}b_{d1} & \sum_{d=1}^{n}a_{2d}b_{d2} & ... & \sum_{d=1}^{n}a_{2d}b_{dn}\\  \\

\end{bmatrix}
